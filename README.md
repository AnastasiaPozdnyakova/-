from  random import choice
word = ("кактус", "котлетка" .split())
x = choice(word).lower()
gueses = ''
turns = 10
while turns > 0:         
    fail = 0               
    for char in x:     
        if char in gueses:    
            print (char,end=""),    
        else:
            print (" # ",end=""),     
            fail += 1    
    if fail == 0:        
        print ("  Поздравляю! ты выйграл")
        break            
    guess = input("угадайте букву:") 
    gueses += guess                    
    if guess not in x:  
        turns -= 1        
        print ("в этом слове нет вашей буквы")  
        print ("У вас осталось", + turns, 'попыток' )
        if turns == 0:           
            print ("Вы проиграли :(")
    if guess == 'дайответблин':
        print("слово:", x)
